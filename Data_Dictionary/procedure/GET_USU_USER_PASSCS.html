CREATE OR REPLACE PROCEDURE GET_USU_USER_PASSCS (
    user_name VARCHAR2,
    pass VARCHAR2,
    CSDATOS OUT SYS_REFCURSOR
) AS
    VMES VARCHAR2(4000);
    VCOD NUMBER;
BEGIN
    OPEN CSDATOS FOR 
        SELECT id_usuario, nombre, apellido, correo, direccion, username, password, tarjeta, pin, fecha, activo, id_rol
        FROM Usuario
        WHERE username = user_name 
        AND password = pass;

EXCEPTION    
    WHEN OTHERS THEN  
        VMES := SQLERRM;
        VCOD := SQLCODE;
        INSERT INTO ERRORES_AUDIT 
        VALUES (USER, 'GET_USU_USER_PASSCS', SYSDATE, VCOD || ' - '|| VMES);
END GET_USU_USER_PASSCS;
