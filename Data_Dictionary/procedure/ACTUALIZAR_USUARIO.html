CREATE OR REPLACE PROCEDURE ACTUALIZAR_USUARIO (
    usuario_id NUMBER,
    nomb VARCHAR2,
    apelli VARCHAR2,
    mail VARCHAR2,
    direc VARCHAR2,
    userna VARCHAR2,
    pass VARCHAR2,
    tarj VARCHAR2,
    pin_us VARCHAR2
) AS
    VMES VARCHAR2(4000);
    VCOD NUMBER;
BEGIN
    UPDATE Usuario
    SET nombre = nomb,
        apellido = apelli,
        correo = mail,
        direccion = direc,
        username = userna,
        password = pass,
        tarjeta = tarj,
        pin = pin_us,
        fecha = SYSDATE,
        activo = 1,
        id_rol = 2
    WHERE id_usuario = usuario_id;

EXCEPTION
    WHEN OTHERS THEN
        VMES := SQLERRM;
        VCOD := SQLCODE;
        INSERT INTO ERRORES_AUDIT
        VALUES (USER, 'ACTUALIZAR_USUARIO', SYSDATE, VCOD || ' - ' || VMES);
END ACTUALIZAR_USUARIO;
