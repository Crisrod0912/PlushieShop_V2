CREATE OR REPLACE FUNCTION GET_USU_USER (
    user_name VARCHAR2
) RETURN SYS_REFCURSOR
AS
    VMES VARCHAR2(4000);
    VCOD NUMBER;
    CDATOS SYS_REFCURSOR;
BEGIN
    GET_USU_USERCS(user_name, CDATOS);
    
    RETURN CDATOS;
    
EXCEPTION
